# 数组去重

使用hasOwnProperty

function unique(arr) {
    var obj = {};
    return arr.filter(function(item, index, arr){
        return obj.hasOwnProperty(typeof item + item) ? false : (obj[typeof item + item] = true)
    })
}
    var arr = [1,1,'true','true',true,true,15,15,false,false, undefined,undefined, null,null, NaN, NaN,'NaN', 0, 0, 'a', 'a',{},{}];
        console.log(unique(arr))
//[1, "true", true, 15, false, undefined, null, NaN, "NaN", 0, "a", {…}]   
//所有的都去重了

# 冒泡排序

function bSort(arr) {
  var len = arr.length;
  for (var i = 0; i < len-1; i++) {
    for (var j = 0; j < len - 1 - i; j++) {
         // 相邻元素两两对比，元素交换，大的元素交换到后面
        if (arr[j] > arr[j + 1]) {
            var temp = arr[j];
            arr[j] = arr[j+1];
            arr[j+1] = temp;
        }
    }
  }
  return arr;
}

# 快速排序

let quickSort = (array) => {
	if (array.length < 2) return array}
	let leftArray = [];
	let rightArray = [];
	let baseDigit = array[0];
	array.forEach(element => {
		if (element < baseDigit) {
			leftArray.push(element);
		} else if (element > baseDigit) {
			rightArray.push(element);
		}
	});
	return quickSort(leftArray).concat(baseDigit, quickSort(rightArray))
};

# 二分查找

function binaryFind(arr, key){
  let low = 0
  let high = arr.length-1
  while(low < high){
    let mid = parseInt((high + low)/2)
    if(key === arr[mid]) return mid
    else if(key > arr[mid]) low = mid + 1
    else if(key < arr[mid]) high = mid - 1
    else return -1
  }
}

# 替换URL指定参数的值

方法一
function replaceParamVal(url,paramName,replaceVal) {
  var oUrl = url.toString();
  var re=eval('/('+ paramName+'=)([^&]*)/gi');
  //修饰符：g全局匹配 i大小写不敏感
  var nUrl = oUrl.replace(re,paramName+'='+replaceVal);
  return nUrl;
}

//方法二
function replaceQueryString (url, name, value) {
  const re = new RegExp(name + '=[^&]*', 'gi')
  return url.replace(re, name + '=' + value)
}

# 找出字符串中第一次只出现一次的字母

String.prototype.firstOnece=function(){

  var len=this.length;
  for(var i=0;i<len;i++){
    var flag=1;
    for(var j=0;j<len;j++){
      if(i==j){
        continue;
      }else{
        if(this[i]==this[j]){
          flag=0;
          break;
        }
      }
    }
    if(flag==1){
      return this[i];
    }
  }
}

# 实现sum(1)(2)()==3

function sum(x){
  return function(y){
    return x+y
  }
}

function sum(...arg1){
    let x = arg1.reduce((prev,next)=>{
        return prev+next
    })
    return function(...arg2){
        if(arg2.length==0) return x
        let y = arg2.reduce((prev,next)=>{
            return prev+next
    })
    return sum(x+y)
    }
}

# base64

  1. 首先取ABC对应的ASCII码值。A（65）B（66）C（67）
  2. 再取二进制值A（01000001）B（01000010）C（01000011）
  3. 然后把这三个字节的二进制码接起来（010000010100001001000011）
  4. 再以6位为单位分成4个数据块,并在最高位填充两个0后形成4个字节的编码后的值，（00010000）（00010100）（00001001）（00000011）
  5. 再把这四个字节数据转化成10进制数得（16）（20）（9）（3）
  6. 最后根据BASE64给出的64个基本字符表，查出对应的ASCII码字符（Q）（U）（J）（D），这里的值实际就是数据在字符表中的索引
